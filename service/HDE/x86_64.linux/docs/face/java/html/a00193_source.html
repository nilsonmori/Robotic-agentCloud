54  private void setupPublisher() {
55  PublisherQos qos = this.getDescription().getPublisherQos();
56  try {
57  if (qos == null) {
58  qos = this.getParticipant().getDefaultPublisherQos();
59  qos.withPolicy(PolicyFactory
60  .getPolicyFactory(this.getDescription().getEnvironment())
61  .Partition().withName(this.getDescription().getName()));
62  } else {
63  Set<String> names = qos.getPartition().getName();
64 
65  if (names.size() == 1 && names.contains("")) {
66  qos = qos.withPolicy(qos.getPartition().withName(
67  this.getDescription().getName()));
68  }
69  }
70  this.publisher = this.getParticipant().createPublisher(qos);
71  this.setLastMessageValidity(VALIDITY_TYPE.VALID);
72  } catch (AlreadyClosedException e) {
73  this.setLastMessageValidity(VALIDITY_TYPE.INVALID);
74  } catch (Exception exc) {
75  Logger.getInstance().log(exc.getMessage(), Level.SEVERE);

106  public RETURN_CODE_TYPE sendMessage(TYPE instanceData, long timeout, LongHolder transaction_id) {
107  try {
108  if (this.dataWriter.getQos().getReliability().getKind() == Reliability.Kind.RELIABLE) {
109  long maxBlockingTime = this.dataWriter.getQos()
110  .getReliability().getMaxBlockingTime()
111  .getDuration(TimeUnit.NANOSECONDS);
112 
113  if (timeout > maxBlockingTime) {
114  Logger.getInstance().log(
115  "Supplied time-out > DataWriterQos.reliability.max_blocking_time ("
116  + timeout + " > " + maxBlockingTime + ").",
117  Level.WARNING);
118 
119  this.setLastMessageValidity(VALIDITY_TYPE.VALID);
120  return RETURN_CODE_TYPE.INVALID_PARAM;
121  }
122  }
123  try {
124  this.dataWriter.write(instanceData);
125  transaction_id.value = this.transactionid.incrementAndGet();
126  this.setLastMessageValidity(VALIDITY_TYPE.VALID);
127  } catch (AlreadyClosedException e) {
128  this.setLastMessageValidity(VALIDITY_TYPE.INVALID);
129  } catch (TimeoutException e) {
130  Logger.getInstance().log("Write time-out", Level.WARNING);
131  this.setLastMessageValidity(VALIDITY_TYPE.VALID);
132  return RETURN_CODE_TYPE.TIMED_OUT;
133  }
134  } catch (AlreadyClosedException e) {
135  this.setLastMessageValidity(VALIDITY_TYPE.INVALID);
136  } catch (Exception exc) {
137  Logger.getInstance().log(exc.getMessage(), Level.SEVERE);
138  Logger.getInstance().log(exc.toString(), Level.FINEST);
139  this.setLastMessageValidity(VALIDITY_TYPE.INVALID);
140  return RETURN_CODE_TYPE.CONNECTION_CLOSED;

142  }
