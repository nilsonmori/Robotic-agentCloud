67  protected Class<? extends T> getClassIdForPolicy(T policy) {
68  Class<?>[] interfaces = policy.getClass().getInterfaces();
69 
70  if (interfaces.length == 0) {
71  throw new IllegalArgumentExceptionImpl(this.environment,
72  "Provided policy not a valid policy.");

74  /*
75  * Ensure extended policies in OpenSplice are still stored using the OMG

79  if (interfaces[0].getName().startsWith(
80  "org.opensplice.dds.core.policy.")) {
81  Class<?>[] superInterfaces = interfaces[0].getInterfaces();
