282  public static DataReaderQosImpl convert(OsplServiceEnvironment env,
283  DDS.DataReaderQos oldQos) {
284  if (oldQos == null) {
285  throw new IllegalArgumentExceptionImpl(env,
286  "oldQos parameter is null.");
287  }
288 
289  DataReaderQosImpl qos = new DataReaderQosImpl(env);
290 
291  qos.put(Deadline.class, PolicyConverter.convert(env, oldQos.deadline));
292  qos.put(DestinationOrder.class,
293  PolicyConverter.convert(env, oldQos.destination_order));
294  qos.put(Durability.class,
295  PolicyConverter.convert(env, oldQos.durability));
296  qos.put(History.class, PolicyConverter.convert(env, oldQos.history));
297  qos.put(LatencyBudget.class,
298  PolicyConverter.convert(env, oldQos.latency_budget));
299  qos.put(Liveliness.class,
300  PolicyConverter.convert(env, oldQos.liveliness));
301  qos.put(Ownership.class, PolicyConverter.convert(env, oldQos.ownership));
302  qos.put(ReaderDataLifecycle.class,
303  PolicyConverter.convert(env, oldQos.reader_data_lifecycle));
304  qos.put(Reliability.class,
305  PolicyConverter.convert(env, oldQos.reliability));
306  qos.put(ResourceLimits.class,
307  PolicyConverter.convert(env, oldQos.resource_limits));
308 
309  qos.put(TimeBasedFilter.class,
310  PolicyConverter.convert(env, oldQos.time_based_filter));
311  qos.put(UserData.class, PolicyConverter.convert(env, oldQos.user_data));
312 
313  Share share = PolicyConverter.convert(env, oldQos.share);
314 
315  if (share != null) {
316  qos.put(Share.class, share);
317  }
318 
319  ReaderLifespan readerLifespan = PolicyConverter.convert(env,
320  oldQos.reader_lifespan);
321 
322  if (readerLifespan != null) {
323  qos.put(ReaderLifespan.class, readerLifespan);
324  }
325  SubscriptionKeys subscriptionKeys = PolicyConverter.convert(env,
326  oldQos.subscription_keys);
327 
328  if (subscriptionKeys != null) {
