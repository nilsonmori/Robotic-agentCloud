217 
218  void operator() (const dds::core::cond::StatusCondition& cond)
219  {
220  const dds::core::status::LivelinessChangedStatus livChangedStatus =
221  ((dds::sub::DataReader<WaitSetData::Msg>&)cond.entity()).liveliness_changed_status();
222 
223  if(livChangedStatus.alive_count() < prevAliveCount)
224  {
225  std::cout << "\n!!! a DataWriter lost its liveliness" << std::endl;
226  writerLeft = true;
227  std::cout << "\n=== Triggering escape condition " << std::endl;
228  guardTriggered = true;
229  }
230  else if(livChangedStatus.alive_count() > prevAliveCount)
231  {
232  std::cout << "\n!!! a DataWriter joined" << std::endl;
233  }
234  prevAliveCount = livChangedStatus.alive_count();
235  }
236 
237 private:
238  int& prevAliveCount;
239  bool& writerLeft;
240  bool& guardTriggered;

273 int subscriber(int argc, char *argv[])
274 {
275  int result = 0;
276  try
277  {
280  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
281  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
282  << dds::core::policy::Durability::Transient()
283  << dds::core::policy::Reliability::Reliable()
284  << dds::core::policy::History::KeepLast(2);
285  dds::topic::Topic<WaitSetData::Msg> topic(dp, "WaitSetData_Msg", topicQos);
286 
288  std::string name = "WaitSet example";
289  dds::sub::qos::SubscriberQos subQos
290  = dp.default_subscriber_qos()
291  << dds::core::policy::Partition(name);
292  dds::sub::Subscriber sub(dp, subQos);
293 
295  dds::sub::qos::DataReaderQos drqos = topic.qos();
296 
298  dds::sub::DataReader<WaitSetData::Msg> dr(sub, topic, drqos);
299 
300  //Logic variables
301  int prevAliveCount = 0;
302  bool closed = false;
303  bool escaped = false;
304  bool writerLeft = false;
305  bool guardTriggered = false;
306  int count = 0;
307 
309  dds::sub::status::DataState newDataState;
310  newDataState << dds::sub::status::SampleState::not_read()
311  << dds::sub::status::ViewState::new_view()
312  << dds::sub::status::InstanceState::any();
313  ReadCondHandler readCondHandler(newDataState);
314  dds::sub::cond::ReadCondition readCond(dr, newDataState, readCondHandler);
315 
317  std::cout << "=== [WaitSetDataSubscriber] Query : message = \"Hello again\"" << std::endl;
318  dds::sub::Query query(dr, "message='Hello again'");
319  dds::sub::status::DataState anyDataState;
320  QueryCondHandler queryCondHandler;
321  dds::sub::cond::QueryCondition queryCond(query, anyDataState, queryCondHandler);
322 
324  StatusCondHandler statusCondHandler(prevAliveCount, writerLeft, guardTriggered);
325  dds::core::cond::StatusCondition statusCond(dr, statusCondHandler);
326  dds::core::status::StatusMask statusMask;
327  statusMask << dds::core::status::StatusMask::liveliness_changed();
328  statusCond.enabled_statuses(statusMask);
329 
331  GuardCondHandler guardCondHandler(escaped, count);
332  dds::core::cond::GuardCondition guardCond;
333  guardCond.handler(guardCondHandler);
334 
336  dds::core::cond::WaitSet waitSet;
337  waitSet += readCond;
338  waitSet += queryCond;
339  waitSet += statusCond;
340  waitSet += guardCond;
341 
342  dds::core::Duration waitTimeout(20, 0);
343 
344  std::cout << "=== [WaitSetDataSubscriber] Ready ..." << std::endl;
345 
346  while(!closed && count < 20)
347  {
349  try
350  {
351  waitSet.dispatch(waitTimeout);
353  guardCond.trigger_value(guardTriggered);
354  }
355  catch(const dds::core::TimeoutError e)
356  {
357  std::cout << std::endl << "!!! [INFO] WaitSet timed out - count = " << count << std::endl;
358  }
359  ++count;
361  closed = escaped && writerLeft;
362  }
363  if(count >= 20)
364  {
365  std::cout << std::endl << "*** Error : Timed out - count = " << count << " ***" << std::endl;
366  result = 1;
367  }
368  else
369  {
370  std::cout << "=== [Subscriber] Closed" << std::endl;
371  }
372  }
373  catch (const dds::core::Exception& e)
374  {
375  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
376  result = 1;
377  }
378  return result;
379 }
