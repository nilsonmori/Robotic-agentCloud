103 
105  const dds::core::status::LivelinessChangedStatus livChangedStatus
106  = chatMessageReader.liveliness_changed_status();
107 
108  if(livChangedStatus.alive_count() < prevAliveCount)
109  {
111  dds::sub::status::DataState notAliveDataState;
112  notAliveDataState << dds::sub::status::SampleState::any()
113  << dds::sub::status::ViewState::any()
114  << dds::sub::status::InstanceState::not_alive_mask();
115  dds::sub::LoanedSamples<Chat::NameService> users
116  = nameServiceReader.select().state(notAliveDataState).take();
117 
119  for (dds::sub::LoanedSamples<Chat::NameService>::const_iterator user = users.begin();
120  user < users.end(); ++user)
121  {
122  if(user->info().valid())

125  std::stringstream queryString;

127  dds::sub::Query query(chatMessageReader, queryString.str());
128  dds::sub::LoanedSamples<Chat::ChatMessage> messages

130 
131  std::cout << "Departed user " << user->data().name()

189 int UserLoad(int argc, char *argv[])
190 {
191  int result = 0;
192  bool terminated = false;
193  try
194  {
196  SubEntities e;
197 
202  dds::sub::status::DataState newDataState;
203  newDataState << dds::sub::status::SampleState::not_read()
204  << dds::sub::status::ViewState::new_view()
205  << dds::sub::status::InstanceState::alive();
206  NewUserHandler newUserHandler;
207  dds::sub::cond::ReadCondition newUser(e.nameServiceReader, newDataState, newUserHandler);
208 
213  UserLeftHandler userLeftHandler(terminated, e.nameServiceReader);
214  dds::core::cond::StatusCondition userLeft(e.chatMessageReader, userLeftHandler);
215  dds::core::status::StatusMask statusMask;
216  statusMask << dds::core::status::StatusMask::liveliness_changed();
217  userLeft.enabled_statuses(statusMask);
218 
223  EscapeHandler escapeHandler(terminated);
224  escape.handler(escapeHandler);
225 
227  dds::core::cond::WaitSet waitSet;
228  waitSet += newUser;
229  waitSet += userLeft;
230  waitSet += escape;
231 
233  dds::sub::status::DataState notAliveDataState;
234  notAliveDataState << dds::sub::status::SampleState::any()
235  << dds::sub::status::ViewState::any()
236  << dds::sub::status::InstanceState::not_alive_mask();
237  e.nameServiceReader.select().state(notAliveDataState).take();
238 
240  os_threadId threadId;
241  os_threadAttr threadAttr;
242  os_threadAttrInit(&threadAttr);
243  os_threadCreate(&threadId, "delayedEscape", &threadAttr, delayedEscape, 0);
244 
245  std::cout << "UserLoad has opened: disconnect a Chatter with userID = "
246  << TERMINATION_MESSAGE << " to close it....\n" << std::endl;
247 
252  while(!terminated)
253  {
254  waitSet.dispatch();
255  }
256  }
257  catch (const dds::core::Exception& e)
258  {
259  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
260  result = 1;
261  }
262  return result;
263 }
