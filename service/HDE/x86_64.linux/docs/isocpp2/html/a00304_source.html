61 void writeSample(std::string writer_action, dds::pub::DataWriter<LifecycleData::Msg> dw)
62 {
64  LifecycleData::Msg msgInstance;
65  if(writer_action == "dispose")
66  {
67  msgInstance.userID() = 1;
68  msgInstance.message() = "Lifecycle_1";
69  msgInstance.writerStates() = "SAMPLE_SENT -> INSTANCE_DISPOSED -> DATAWRITER_DELETED";
70  }
71  else if(writer_action == "unregister")
72  {
73  msgInstance.userID() = 2;
74  msgInstance.message() = "Lifecycle_2";
75  msgInstance.writerStates() = "SAMPLE_SENT -> INSTANCE_UNREGISTERED -> DATAWRITER_DELETED";
76  }
77  else if(writer_action == "stoppub")
78  {
79  msgInstance.userID() = 3;
80  msgInstance.message() = "Lifecycle_3";
81  msgInstance.writerStates() = "SAMPLE_SENT -> DATAWRITER_DELETED";
82  }
83 
84  std::cout << "=== [Publisher] written a message containing :" << std::endl;
85  std::cout << " userID : " << msgInstance.userID() << std::endl;
86  std::cout << " Message : \"" << msgInstance.message() << "\"" << std::endl;
87  std::cout << " writerStates : \"" << msgInstance.writerStates() << "\"" << std::endl;
88  dw << msgInstance;
89  exampleSleepMilliseconds(500);
90  std::cout << "=== [Publisher] : SAMPLE_SENT" << std::endl;
91 
93  if(writer_action == "dispose")
94  {
95  dw.dispose_instance(dw.lookup_instance(msgInstance));
96  }
97  else if(writer_action == "unregister")
98  {
99  dw.unregister_instance(dw.lookup_instance(msgInstance));
100  }

107 int publisher(int argc, char *argv[])
108 {
109  int result = 0;
110 
111  if(argc < 3 ||
112  ((strcmp(argv[1], "false") != 0) &&
113  (strcmp(argv[1], "true") != 0) &&
114  (strcmp(argv[2], "dispose") != 0) &&
115  (strcmp(argv[2], "unregister") != 0) &&
116  (strcmp(argv[2], "stoppub") != 0)))
117  {
118  usagePub();
119  return -1;
120  }
121  bool autodispose_unregistered_instances = (strcmp(argv[1], "true") == 0);
122  std::string writer_action = argv[2];
123 
124  try
125  {
127  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
128 
133  dds::topic::qos::TopicQos topicQos
134  = dp.default_topic_qos()
135  << dds::core::policy::Durability::Transient()
136  << dds::core::policy::Reliability::Reliable();
137 
139  dds::topic::Topic<LifecycleData::Msg> topic(dp, "LifecycleData_Msg", topicQos);
140 
142  std::string name = "Lifecycle example";
143  dds::pub::qos::PublisherQos pubQos
144  = dp.default_publisher_qos()
145  << dds::core::policy::Partition(name);
146  dds::pub::Publisher pub(dp, pubQos);
147 
153  dds::pub::qos::DataWriterQos dwqos = topic.qos();
154  dwqos << dds::core::policy::WriterDataLifecycle(autodispose_unregistered_instances);
155 
158  dds::pub::DataWriter<LifecycleData::Msg> dw(pub, topic, dwqos);
159  dds::pub::DataWriter<LifecycleData::Msg> dwStopper(pub, topic, dwqos);
160  exampleSleepMilliseconds(500);
161 
162  writeSample(writer_action, dw);
163 
164  std::cout << "=== [Publisher] waiting 500ms to let the subscriber handle the previous write state ..." << std::endl;
165  exampleSleepMilliseconds(500);
166 
168  dw = dds::pub::DataWriter<LifecycleData::Msg>(dds::core::null);
169  exampleSleepMilliseconds(500);
170 
172  LifecycleData::Msg msgInstance(4, "Lifecycle_4", "STOPPING_SUBSCRIBER");
173  std::cout << "=== [Publisher] :" << std::endl;
174  std::cout << " userID : " << msgInstance.userID() << std::endl;
175  std::cout << " Message : \"" << msgInstance.message() << "\"" << std::endl;
176  std::cout << " writerStates : \"" << msgInstance.writerStates() << "\"" << std::endl;
177  dwStopper << msgInstance;
178 
179  /* A sleep ensures time is allowed for the sample to be written to the network.
180  If the example is running in *Single Process Mode* exiting immediately might
181  otherwise shutdown the domain services before this could occur */
182  exampleSleepMilliseconds(1000);
183  }
184  catch (const dds::core::Exception& e)
185  {
186  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
187  result = 1;
188  }
189  return result;
190 }

250 int subscriber(int argc, char *argv[])
251 {
252  int result = 0;
253  try
254  {
257  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
258  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
259  << dds::core::policy::Durability::Transient()
260  << dds::core::policy::Reliability::Reliable();
261  dds::topic::Topic<LifecycleData::Msg> topic(dp, "LifecycleData_Msg", topicQos);
262 
264  std::string name = "Lifecycle example";
265  dds::sub::qos::SubscriberQos subQos
266  = dp.default_subscriber_qos()
267  << dds::core::policy::Partition(name);
268  dds::sub::Subscriber sub(dp, subQos);
269 
271  dds::sub::qos::DataReaderQos drqos = topic.qos();
272 
274  dds::sub::DataReader<LifecycleData::Msg> dr(sub, topic, drqos);
275 
276  std::cout << "=== [Subscriber] Ready ..." << std::endl;
277 
280  bool closed = false;
281  int count = 0;
282  int maxCount = 100;
283  do
284  {
285  dds::sub::LoanedSamples<LifecycleData::Msg> samples = dr.read();
286  for (dds::sub::LoanedSamples<LifecycleData::Msg>::const_iterator sample = samples.begin();
287  sample < samples.end();
288  ++sample)
289  {
290  std::cout << "=== [Subscriber] message received :" << std::endl;
291  std::cout << " userID : " << sample->data().userID() << std::endl;
292  std::cout << " Message : \"" << sample->data().message() << "\"" << std::endl;
293  std::cout << " writerStates : \"" << sample->data().writerStates() << "\"" << std::endl;
294  std::cout << " valid_data : " << sample->info().valid() << std::endl;
295  std::cout << " sample_state:" << sampleStateToString(sample->info().state().sample_state());
296  std::cout << "-view_state:" << viewStateToString(sample->info().state().view_state());
297  std::cout << "-instance_state:" << instanceStateToString(sample->info().state().instance_state()) << std::endl;
298  closed = (sample->data().writerStates() == "STOPPING_SUBSCRIBER");
299  ++count;
300  }
301  exampleSleepMilliseconds(200);
302  }
303  while (!closed && count < maxCount);
304 
305  std::cout << "=== [Subscriber] stopping after " << count << " iterations - closed = " << closed << std::endl;
306  if (count == maxCount)
307  {
308  std::cout << "*** Error : max " << maxCount << " iterations reached" << std::endl;
309  result = 1;
310  }
311  }
312  catch (const dds::core::Exception& e)
313  {
314  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
315  result = 1;
316  }
317  return result;
318 }
