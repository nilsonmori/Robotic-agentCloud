146  = dp.default_subscriber_qos()
147  << dds::core::policy::Partition(name);

149  dds::sub::qos::DataReaderQos drqos = topic.qos();
150 
154  ExampleDataReaderListener listener;
155  dds::core::status::StatusMask mask;
156  mask << dds::core::status::StatusMask::data_available()
157  << dds::core::status::StatusMask::requested_deadline_missed();
158  std::cout << "=== [ListenerDataSubscriber] Set listener" << std::endl;
159  dds::sub::DataReader<ListenerData::Msg> dr(sub, topic, drqos, &listener, mask);
160  std::cout << "=== [ListenerDataSubscriber] Ready ..." << std::endl;
161 
165  int count = 0;
166  while(! listener.deadline_expired_ && count < 20)
167  {
168  exampleSleepMilliseconds(1000);
169  count++;
170  }
171 
172  result = ! (listener.deadline_expired_ && listener.data_received_);
173 
175  dr.listener(0, dds::core::status::StatusMask::none());
176 
177  std::cout << "=== [ListenerDataSubscriber] Closed" << std::endl;
178  }
179  catch (const dds::core::Exception& e)
180  {
181  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
182  result = 1;
183  }
