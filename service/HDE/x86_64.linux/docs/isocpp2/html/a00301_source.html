65 int publisher(int argc, char *argv[])
66 {
67  int result = 0;
68  try
69  {
71  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
72 
77  dds::topic::qos::TopicQos topicQos
78  = dp.default_topic_qos()
79  << dds::core::policy::Durability::Transient()
80  << dds::core::policy::Reliability::Reliable()
81  << dds::core::policy::DestinationOrder::SourceTimestamp();
82 
84  dds::topic::Topic<StockMarket::Stock> topic(dp, "StockTrackerExclusive", topicQos);
85 
87  std::string name = "ContentFilteredTopic example";
88  dds::pub::qos::PublisherQos pubQos
89  = dp.default_publisher_qos()
90  << dds::core::policy::Partition(name);
91  dds::pub::Publisher pub(dp, pubQos);
92 
99  dds::pub::qos::DataWriterQos dwqos = topic.qos();
100  dwqos << dds::core::policy::WriterDataLifecycle::AutoDisposeUnregisteredInstances();
101 
103  dds::pub::DataWriter<StockMarket::Stock> dw(pub, topic, dwqos);
104 
106  StockMarket::Stock geQuote("GE", 12.00f);
107 
108  StockMarket::Stock msftQuote("MSFT", 25.00f);
109 
111  for (unsigned int i = 0; i < write_loop_count; i++)
112  {
113  geQuote.price() += 0.5f;
114  msftQuote.price() += 1.5f;
115  std::cout << "=== [ContentFilteredTopicDataPublisher] sends 2 stockQuotes : (GE, "
116  << geQuote.price() << ") (MSFT, " << msftQuote.price() << ")" << std::endl;
117  dw << geQuote;
118  dw << msftQuote;
119  exampleSleepMilliseconds(1000);
120  }
122  geQuote.price() = -1;
123  msftQuote.price() = -1;
124  dw << geQuote;
125  dw << msftQuote;
126  /* A short sleep ensures time is allowed for the sample to be written to the network.
127  If the example is running in *Single Process Mode* exiting immediately might
128  otherwise shutdown the domain services before this could occur */
129  exampleSleepMilliseconds(1000);
130  }
131  catch (const dds::core::Exception& e)
132  {
133  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
134  result = 1;
135  }
136  return result;
137 }

143 int subscriber(int argc, char *argv[])
144 {
145  int result = 0;
146  try
147  {
151  if(argc < 2)
152  {
153  std::cerr << "*** [ContentFilteredTopicDataSubscriber] Subscription string not specified" << std::endl;
154  std::cerr << "*** usage : ContentFilteredTopicDataSubscriber <subscription_string>" << std::endl;
155  return - 1;
156  }
157 
158  std::string requested_ticker = argv[1];
159 
162  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
163  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
164  << dds::core::policy::Durability::Transient()
165  << dds::core::policy::Reliability::Reliable()
166  << dds::core::policy::DestinationOrder::SourceTimestamp();
167  dds::topic::Topic<StockMarket::Stock> topic(dp, "StockTrackerExclusive", topicQos);
168 
170  std::string name = "ContentFilteredTopic example";
171  dds::sub::qos::SubscriberQos subQos
172  = dp.default_subscriber_qos()
173  << dds::core::policy::Partition(name);
174  dds::sub::Subscriber sub(dp, subQos);
175 
177  dds::sub::qos::DataReaderQos drqos = topic.qos();
178 
179 
182  std::stringstream ss;
183  ss << "ticker = '" << requested_ticker << "'";
184  dds::topic::Filter filter(ss.str());
185  dds::topic::ContentFilteredTopic<StockMarket::Stock> cftopic(topic, "CFStockTrackerExclusive", filter);
186 
188  dds::sub::DataReader<StockMarket::Stock> dr(sub, cftopic, drqos);
189 
192  bool terminate_sample_received = false;
193  unsigned int correct_quote_count = 0;
194  for (unsigned int count = 0; count < write_loop_count * 10 && !terminate_sample_received; ++count)
195  {
196  dds::sub::LoanedSamples<StockMarket::Stock> samples = dr.take();
197  for (dds::sub::LoanedSamples<StockMarket::Stock>::const_iterator sample = samples.begin();
198  sample < samples.end();
199  ++sample)
200  {
201  if(sample->info().valid())
202  {
203  if (sample->data().price() == -1)
204  {
205  terminate_sample_received = true;
206  break;
207  }
208  std::cout << "=== [ContentFilteredTopicDataSubscriber] receives stockQuote : ("
209  << sample->data().ticker() << ", " << sample->data().price() << ")" << std::endl;
210 
211  if (requested_ticker == sample->data().ticker())
212  ++correct_quote_count;
213  else
214  {
215  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Unexpected quote received for "
216  << sample->data().ticker() << std::endl;
217  result = 1;
218  }
219  }
220  }
221  exampleSleepMilliseconds(900);
222  }
223 
225  if (terminate_sample_received)
226  {
227  std::cout << "=== [ContentFilteredTopicDataSubscriber] Market Closed." << std::endl;
228  if (correct_quote_count != (requested_ticker == "GE" || requested_ticker == "MSFT" ? write_loop_count : 0))
229  {
230  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Incorrect quote count." << std::endl;
231  result = 1;
232  }
233  }
234  else
235  {
236  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Timed out waiting for close signal." << std::endl;
237  result = 1;
238  }
239  }
240  catch (const dds::core::Exception& e)
241  {
242  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
243  result = 1;
244  }
245  return result;
246 }

248 }
249 }
