413  int result = 0;
414 
415  try
416  {
418  Entities e("pong", "ping");
419 
420  std::cout << "Waiting for samples from ping to send back..." << std::endl;
421 
422  dds::core::cond::WaitSet::ConditionSeq conditions;
423  while(!terminated.trigger_value())
424  {
426  e.waitSet.wait(conditions);
427 
429  dds::sub::LoanedSamples<RoundTripModule::DataType> samples = e.reader.take();
430  for(dds::sub::LoanedSamples<RoundTripModule::DataType>::const_iterator sample = samples.begin();
431  !terminated.trigger_value() && sample < samples.end();
432  ++sample)
433  {
435  if(sample->info().state().instance_state() == dds::sub::status::InstanceState::not_alive_disposed())
436  {
437  std::cout << "Received termination request. Terminating." << std::endl;
438  terminated.trigger_value(true);
439  break;
440  }
442  else if(sample->info().valid())
443  {
444  e.writer << sample->data();
445  }
446  }
447  }
448  }
449  catch(const dds::core::Exception& e)
450  {
451  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
452  result = 1;
453  }
454 
455 #ifdef _WIN32
456  SetConsoleCtrlHandler(0, false);
457 #else
458  sigaction(SIGINT,&oldAction, 0);
