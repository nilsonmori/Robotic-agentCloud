66 int publisher(int argc, char *argv[])
67 {
68  int result = 0;
69 
70  if(argc < 5 || (strcmp(argv[4], "0") && strcmp(argv[4], "1")))
71  {
72  usagePub();
73  return -1;
74  }
75  std::string publisher_name = argv[1];
76  int ownership_strength = atoi(argv[2]);
77  int max_iterations = atoi(argv[3]);
78  bool stop_subscriber_flag = (strcmp(argv[4], "1") == 0);
79 
80  try
81  {
83  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
84 
89  dds::topic::qos::TopicQos topicQos
90  = dp.default_topic_qos()
91  << dds::core::policy::Durability::Transient()
92  << dds::core::policy::Reliability::Reliable()
93  << dds::core::policy::Ownership::Exclusive();
94 
96  dds::topic::Topic<OwnershipData::Stock> topic(dp, "OwnershipStockTracker", topicQos);
97 
99  std::string name = "Ownership example";
100  dds::pub::qos::PublisherQos pubQos
101  = dp.default_publisher_qos()
102  << dds::core::policy::Partition(name);
103  dds::pub::Publisher pub(dp, pubQos);
104 
111  dds::pub::qos::DataWriterQos dwqos = topic.qos();
112  dwqos << dds::core::policy::WriterDataLifecycle::ManuallyDisposeUnregisteredInstances()
113  << dds::core::policy::OwnershipStrength(ownership_strength);
114 
116  dds::pub::DataWriter<OwnershipData::Stock> dw(pub, topic, dwqos);
117 
118  std::cout << "=== [Publisher] Publisher " << publisher_name << " with strength : " << ownership_strength;
119  std::cout << " / sending " << max_iterations << " prices ..." << " stop_subscriber_flag=" << argv[4] << std::endl;
120 
122  OwnershipData::Stock msgInstance("MSFT", 10.0f, publisher_name.c_str(), ownership_strength);
123  dds::core::InstanceHandle ih = dw.register_instance(msgInstance);
124 
125  for (int i = 0; i < max_iterations; i++)
126  {
127  dw.write(msgInstance, ih);
128  exampleSleepMilliseconds(200);
129  msgInstance.price() += 0.5f;
130  }
131  exampleSleepMilliseconds(2000);
132 
134  if (stop_subscriber_flag)
135  {
136  msgInstance.price() = -1.0f;
137  std::cout << "=== Stopping the subscriber" << std::endl;
138  dw.write(msgInstance, ih);
139  }
140 
141  /* A short sleep ensures time is allowed for the sample to be written to the network.
142  If the example is running in *Single Process Mode* exiting immediately might
143  otherwise shutdown the domain services before this could occur */
144  exampleSleepMilliseconds(1000);
145  }
146  catch (const dds::core::Exception& e)
147  {
148  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
149  result = 1;
150  }
151  return result;
152 }

158 int subscriber(int argc, char *argv[])
159 {
160  int result = 0;
161  try
162  {
165  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
166  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
167  << dds::core::policy::Durability::Transient()
168  << dds::core::policy::Reliability::Reliable()
169  << dds::core::policy::Ownership::Exclusive();
170  dds::topic::Topic<OwnershipData::Stock> topic(dp, "OwnershipStockTracker", topicQos);
171 
173  std::string name = "Ownership example";
174  dds::sub::qos::SubscriberQos subQos
175  = dp.default_subscriber_qos()
176  << dds::core::policy::Partition(name);
177  dds::sub::Subscriber sub(dp, subQos);
178 
180  dds::sub::qos::DataReaderQos drqos = topic.qos();
181 
183  dds::sub::DataReader<OwnershipData::Stock> dr(sub, topic, drqos);
184 
185  std::cout << "===[Subscriber] Ready ..." << std::endl;
186  std::cout << " Ticker Price Publisher ownership strength" << std::endl;
187 
190  bool closed = false;
191  int count = 0;
192  do
193  {
194  dds::sub::LoanedSamples<OwnershipData::Stock> samples = dr.take();
195  for (dds::sub::LoanedSamples<OwnershipData::Stock>::const_iterator sample = samples.begin();
196  sample < samples.end();
197  ++sample)
198  {
199  if (sample->info().valid())
200  {
201  if(sample->data().price() < -0.0f)
202  {
203  closed = true;
204  break;
205  }
206  std::cout << " " << sample->data().ticker() << " " << sample->data().price() << " " << sample->data().publisher() << " " << sample->data().strength() << std::endl;
207  }
208  }
209  exampleSleepMilliseconds(200);
210  ++count;
211  }
212  while (!closed && count < 1500);
213 
214  std::cout << "===[Subscriber] Market Closed" << std::endl;
215  }
216  catch (const dds::core::Exception& e)
217  {
218  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
219  result = -1;
220  }
221  return result;
222 }
