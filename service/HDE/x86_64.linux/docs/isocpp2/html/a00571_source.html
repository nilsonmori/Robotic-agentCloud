1 ISO C++ example code snippets {#snippets}
2 =============================
3 
4 Below you will find an assortment of small code examples which show you how to do various
5 things with the ISO C++ API. Most of the snippets are written in a way that they would be compilable
6 as one program and as such concepts already demonstrated are not repeated in subsequent snippets.
7 
8 Using the DataWriter {#datawriter}
9 ====================
10 
11 Creating a DataWriter with default QoS
12 --------------------------------------
13 ~~~~~~~~~~~~~~~{.cpp}
14  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
15  dds::pub::Publisher pub(dp);
16  dds::topic::Topic<Foo::Type> topic(dp, "TypeTopic");
17  dds::pub::DataWriter<Foo::Type> writer(pub, topic);
18 ~~~~~~~~~~~~~~~
19 Creating a DataWriter with your own QoS
20 ---------------------------------------
21 ~~~~~~~~~~~~~~~{.cpp}
22  dds::pub::qos::DataWriterQos dwqos = topic.qos();
23  dwqos << dds::core::policy::WriterDataLifecycle::ManuallyDisposeUnregisteredInstances();
24  dds::pub::DataWriter<Foo::Type> writer2(pub, topic, dwqos);
25 ~~~~~~~~~~~~~~~
26 Getting and setting DataWriter QoS
27 ----------------------------------
28 ~~~~~~~~~~~~~~~{.cpp}
29  writer >> dwqos;
30  dwqos << dds::core::policy::Deadline(dds::core::Duration::from_microsecs(10))
31  << dds::core::policy::Durability::Volatile();
32 
33  writer << dwqos;
34  //or
35  writer.qos(dwqos);
36 ~~~~~~~~~~~~~~~
37 Writing a sample
38 ----------------
39 ~~~~~~~~~~~~~~~{.cpp}
40  Foo::Type sample;
41  sample.long_1()=1;
42  sample.long_2()=2;
43 
44  writer << sample;
45  //or
46  writer.write(sample);
47 ~~~~~~~~~~~~~~~
48 Writing with other items (timestamps, instance handles)
49 -------------------------------------------------------
50 ~~~~~~~~~~~~~~~{.cpp}
51  dds::core::Time t1(20,0);
52  writer.write(sample, t1);
53  //or
54  writer << std::make_pair(sample, t1);
55 ~~~~~~~~~~~~~~~
56 Writing from your own container
57 -------------------------------
58 ~~~~~~~~~~~~~~~{.cpp}
59  std::vector<Foo::Type> userSamples;
60  userSamples.push_back(sample);
61  writer.write(userSamples.begin(), userSamples.end());
62 ~~~~~~~~~~~~~~~
63 Writing from your own samples container with your own instance handles
64 ----------------------------------------------------------------------
65 ~~~~~~~~~~~~~~~{.cpp}
66  std::vector<dds::core::InstanceHandle> handles;
67  handles.push_back(writer.register_instance(sample));
68  writer.write(userSamples.begin(), userSamples.end(), handles.begin(), handles.end());
69 ~~~~~~~~~~~~~~~
70 Register and instance handle and a write sample using it
71 --------------------------------------------------------
72 ~~~~~~~~~~~~~~~{.cpp}
73  dds::core::InstanceHandle ih(writer.register_instance(sample));
74  writer << std::make_pair(sample, ih);
75 ~~~~~~~~~~~~~~~
76 Using the DataReader {#datareader}
77 ====================
78 
79 Creating a DataReader with the default QoS
80 ------------------------------------------
81 ~~~~~~~~~~~~~~~{.cpp}
82  dds::sub::Subscriber sub(dp);
83  dds::sub::DataReader<Foo::Type> reader(sub, topic);
84 ~~~~~~~~~~~~~~~
85 Creating a DataReader with your own QoS
86 ---------------------------------------
87 ~~~~~~~~~~~~~~~{.cpp}
88  dds::sub::qos::DataReaderQos drqos(topic.qos());
89  drqos << dds::core::policy::ResourceLimits(1000);
90  dds::sub::DataReader<Foo::Type> reader2(sub, topic, drqos);
91 ~~~~~~~~~~~~~~~
92 Reading a sample
93 ----------------
94 ~~~~~~~~~~~~~~~{.cpp}
95  dds::sub::LoanedSamples<Foo::Type> samples = reader.read();
96  //Or
97  reader >> samples;
98 ~~~~~~~~~~~~~~~
99 Reading samples into your own containers (Forward inserting Iterator)
100 ---------------------------------------------------------------------
101 ~~~~~~~~~~~~~~~{.cpp}
102  std::vector<dds::sub::Sample<Foo::Type> > mySamples;
103  mySamples.resize(30);
104  unsigned int numRead = reader.read(mySamples.begin(), 30);
105 ~~~~~~~~~~~~~~~
106 Reading samples into your own containers (Back inserting Iterator)
107 ------------------------------------------------------------------
108 ~~~~~~~~~~~~~~~{.cpp}
109  numRead = reader.read(back_inserter(mySamples));
110 ~~~~~~~~~~~~~~~
111 Reading with a InstanceHandle (lookup key from sample)
112 ------------------------------------------------------
113 ~~~~~~~~~~~~~~~{.cpp}
114  ih = reader.lookup_instance(sample);
115  samples = reader.select().
116  instance(ih).
117  state(dds::sub::status::DataState::any_data())
118  .read();
119 ~~~~~~~~~~~~~~~
120 Reading samples with the Selector
121 ---------------------------------
122 ~~~~~~~~~~~~~~~{.cpp}
123  dds::sub::LoanedSamples<Foo::Type> FooSamples = reader.select()
124  .instance(ih)
125  .state(dds::sub::status::DataState::any_data())
126  .read();
127 ~~~~~~~~~~~~~~~
128 Reading samples with the Manipulator Selector
129 ---------------------------------------------
130 ~~~~~~~~~~~~~~~{.cpp}
131  reader >> dds::sub::instance(ih) >> dds::sub::state(dds::sub::status::DataState::any_data()) >> samples;
132 
133  //Take
134  reader.take();
135  //Or with the ManipulatorSelector
136  reader >> dds::sub::take >> samples;
137 ~~~~~~~~~~~~~~~
138 Accessing the samples
139 ---------------------
140 ~~~~~~~~~~~~~~~{.cpp}
141  for (dds::sub::LoanedSamples<Foo::Type>::const_iterator sample = samples.begin();
142  sample < samples.end(); ++sample)
143  {
144  if (sample->info().valid())
145  {
146  std::cout << sample->data().long_1() << std::endl;
147  }
148  }
149 ~~~~~~~~~~~~~~~
150 Accessing the samples (C++11)
151 -----------------------------
152 ~~~~~~~~~~~~~~~{.cpp}
153  for(auto s : reader.read())
154  {
155  std::cout << s->data().long_1() << std::endl;
156  }
157 ~~~~~~~~~~~~~~~
158 Accessing the samples (C++11 lambda)
159 ------------------------------------
160 ~~~~~~~~~~~~~~~{.cpp}
161  auto s = reader.read();
162  std::for_each(s.begin(), s.end(), [&]{ std::cout << s->data().long_1() << std::endl; });

164 Miscellaneous {#miscellaneous}

166 Getting QoS values
167 ----------------------
168 ~~~~~~~~~~~~~~~{.cpp}
169  dds::core::Duration lease_duration = drqos.policy<dds::core::policy::Liveliness>().lease_duration();
170 ~~~~~~~~~~~~~~~
171 Finding entities (retained or built in)
172 --------------------------------------
173 ~~~~~~~~~~~~~~~{.cpp}
174  //Finding one DataReader (Forward inserting iterator)
175  std::vector<dds::sub::AnyDataReader> rv;
176  rv.push_back(dds::sub::AnyDataReader(dds::core::null));
177  dds::sub::find<dds::sub::DataReader<Foo::Type> >(sub, topic.name(), rv.begin(), 1);
178  samples = dds::sub::DataReader<Foo::Type>(rv[0]).take();
179 
180  //Finding all the DataReaders (Back inserting iterator)
181  dds::sub::find<dds::sub::DataReader<Foo::Type> >(sub, topic.name(), back_inserter(rv));
182  dds::sub::DataReader<Foo::Type> adr(rv[0]);
183  adr >> dds::sub::take >> samples;
184 ~~~~~~~~~~~~~~~
