94 NodeInfo& getNodeInfo(const dds::topic::ParticipantBuiltinTopicData& participantData)
95 {
97  for(std::vector<NodeInfo>::iterator it = nodes.begin(); it < nodes.end(); it++)
98  {
99  if(it->nodeID_ == participantData.key().value()[0])
100  {
102  if(participantData.user_data().value().size() > 0
103  && *participantData.user_data().value().data() != '<' )
104  {
105  it->hostName_ = std::string(reinterpret_cast<const char*> (participantData.user_data().value().data()),
106  participantData.user_data().value().size());
107  }
108  return *it;
109  }
110  }

113  NodeInfo node(participantData);

150  }
151  else

155 
156  std::cout << "=== [BuiltInTopicsDataSubscriber] Waiting for historical data ... " << std::endl;
157 
158  /* Make sure all historical data is delivered in the DataReader */

161  std::cout << "=== [BuiltInTopicsDataSubscriber] Done" << std::endl;

163  /* Create a new ReadCondition for the reader that matches all samples.*/

201  if(nodeInfo.participants.size() == 1)
